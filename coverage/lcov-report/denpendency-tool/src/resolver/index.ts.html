import * as path from 'path';
import { resolve } from 'url';
const l = require('debug')('resolve');
import * as ts from 'typescript';
export default function() {}
 
 
/**
 * ts 自身实现了类似node的resolver,但是增加了d.ts等类型的搜索
 * @param partial the dependency path
 * @param filename the path to all files
 * @param directory  the path to the file containing the partial
 */
export function tsReolve(partial, filename, directory) {
  const options = {
    module: ts.ModuleKind.AMD,
  };
 
  const host = ts.createCompilerHost({});
 
  const resolvedModule = ts.resolveModuleName(partial, filename, options, host)
    .resolvedModule;
  l('ts resolved module: ', resolvedModule);
  const result = resolvedModule ? resolvedModule.resolvedFileName : '';
  l('result: ' + result);
  return result ? path.resolve(result) : '';
}
